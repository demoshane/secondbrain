   ```python
   # Replace the LIKE scan with junction table lookups
   sql = """
       SELECT DISTINCT n.path, n.title, n.type, n.body, n.created_at, n.tags, n.people
       FROM notes n
       JOIN note_people np ON np.note_path = n.path
       WHERE np.person LIKE ?
       UNION
       SELECT DISTINCT n.path, n.title, n.type, n.body, n.created_at, n.tags, n.people
   ```
