        from engine.adapters.claude_adapter import ClaudeAdapter
        import subprocess
        adapter = ClaudeAdapter()
        mock_result = MagicMock()
        mock_result.returncode = 0
        mock_result.stdout = "  response text  "
        with patch("subprocess.run", return_value=mock_result) as mock_run:

        assert result == "response text"
        # Verify user content is not in system_prompt position
        call_args = mock_run.call_args[0][0]
        assert "user content" in call_args[2]  # appears in the combined prompt arg
