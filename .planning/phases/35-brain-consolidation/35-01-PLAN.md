    1. Read keep note: `SELECT path, title, body, tags FROM notes WHERE path = ?`
    2. Read discard note: same query for discard_path. Raise ValueError if either not found.
    3. Merge body: if keep_body and discard_body both non-empty, `merged = keep_body + "\n\n---\n\n" + discard_body`; elif only discard has body, use discard body; else keep body.
    4. Merge tags: `json.loads(keep_tags or '[]')` union `json.loads(discard_tags or '[]')` → `json.dumps(sorted(set(keep_list + discard_list)))`.
    5. In single transaction:
       - `UPDATE notes SET body=?, tags=?, updated_at=datetime('now') WHERE path=?` (keep_path)
       - Remap relationships: for each row where source_path=discard_path, `UPDATE relationships SET source_path=? WHERE source_path=? AND target_path NOT IN (SELECT target_path FROM relationships WHERE source_path=?)` (keep, discard, keep). Then same for target_path. Then `DELETE FROM relationships WHERE source_path=? OR target_path=?` (discard, discard).
       - `DELETE FROM note_embeddings WHERE note_path=?` (discard)
       - `DELETE FROM action_items WHERE note_path=?` (discard) — actions on discard are considered duplicates of keep's actions
       - `DELETE FROM note_people WHERE note_path=?` (discard)
       - `DELETE FROM note_tags WHERE note_path=?` (discard)
       - `DELETE FROM notes WHERE path=?` (discard)
       - `conn.execute("INSERT INTO notes_fts(notes_fts) VALUES('rebuild')")`
    6. Commit.
    7. Delete discard file from disk: `Path(discard_path).unlink(missing_ok=True)`. If path is relative, resolve via `engine.paths.BRAIN_ROOT / discard_path`.
    8. Audit log: `INSERT INTO audit_log (event_type, note_path, detail, created_at) VALUES ('merge', ?, ?, datetime('now'))` with note_path=keep_path, detail=f"merged:{discard_path}".
    9. Commit audit.
    10. Return `{"keep": keep_path, "discarded": discard_path, "merged_tags": merged_tags_list}`.

    ```python
    @mcp.tool()
    def sb_merge_confirm(keep_path: str, discard_path: str, confirm_token: str = "") -> dict:
        """Merge two duplicate notes. Requires confirm_token (two-step safety)."""
        if not confirm_token:
            tok = _issue_token()
            return {"status": "pending", "confirm_token": tok,
                    "message": f"Will merge '{discard_path}' into '{keep_path}'. Call again with confirm_token='{tok}' within 60s."}
        if not _consume_token(confirm_token):
            raise ValueError("TOKEN_EXPIRED: confirm_token invalid or expired.")
        from engine.brain_health import merge_notes
        conn = get_connection()
        try:
            result = merge_notes(keep_path, discard_path, conn)
            return {"status": "merged", **result}
        finally:
            conn.close()
    ```

    **CLI entry point (engine/merge_cli.py) — NEW FILE per D-01:**

    ```python
    """Interactive CLI for merging duplicate notes.

    Entry point: sb-merge-duplicates (registered in pyproject.toml).
    Prompts user through each duplicate candidate pair, calls merge_notes().
    """
    def merge_duplicates_main() -> None:
        from engine.db import get_connection, init_schema
        from engine.brain_health import get_duplicate_candidates, merge_notes

        conn = get_connection()
        init_schema(conn)
        try:
            pairs = get_duplicate_candidates(conn)
            if not pairs:
                print("No duplicate candidates found.")
                return
            print(f"Found {len(pairs)} duplicate candidate(s).\n")
            for i, pair in enumerate(pairs, 1):
                print(f"--- Pair {i}/{len(pairs)} (similarity: {pair['similarity']:.2%}) ---")
                print(f"  A: {pair['a']}")
                print(f"  B: {pair['b']}")
                choice = input("  Merge? [a=keep A, b=keep B, s=skip]: ").strip().lower()
                if choice == "a":
                    result = merge_notes(pair["a"], pair["b"], conn)
                    print(f"  Merged: kept {result['keep']}, discarded {result['discarded']}")
                elif choice == "b":
                    result = merge_notes(pair["b"], pair["a"], conn)
                    print(f"  Merged: kept {result['keep']}, discarded {result['discarded']}")
                else:
                    print("  Skipped.")
            print("\nDone.")
        finally:
            conn.close()
    ```

    ```python
    def brain_health_merge():

        data = request.get_json(force=True)
        keep = data.get("keep_path")
        discard = data.get("discard_path")
        if not keep or not discard:
            return jsonify({"error": "keep_path and discard_path required"}), 400
        conn = get_connection()
        try:
            from engine.brain_health import merge_notes
            result = merge_notes(keep, discard, conn)
            return jsonify(result)
        except ValueError as e:
    ```
