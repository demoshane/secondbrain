    Add at the end of background.js (after existing context menu and Gmail handlers):
    ```javascript
    // --- API Status Badge Polling ---
    async function checkApiStatus() {
      const apiUrl = await getApiUrl();
      try {
        const res = await fetch(`${apiUrl}/ping`, {
          signal: AbortSignal.timeout(2000)
        });
        if (res.ok) {
          chrome.action.setBadgeText({ text: '' });
        } else {
          setOfflineBadge();
        }
      } catch {
        setOfflineBadge();
      }
    }

    function setOfflineBadge() {
      chrome.action.setBadgeText({ text: '!' });
      chrome.action.setBadgeBackgroundColor({ color: '#cc0000' });
    }

    // Use chrome.alarms for reliable background polling (setInterval does NOT
    // persist across service worker sleep cycles)
    chrome.alarms.create('check-api-status', { periodInMinutes: 0.5 }); // 30 seconds
    chrome.alarms.onAlarm.addListener((alarm) => {
      if (alarm.name === 'check-api-status') {
        checkApiStatus();
      }
    });

    // Also check on service worker wake and install
    chrome.runtime.onStartup.addListener(checkApiStatus);
    checkApiStatus(); // initial check
    ```

    **popup.js — capture history rendering (per D-11):**

    Add/update these functions:
    ```javascript
    const MAX_HISTORY = 10;

    function addToHistory(entry) {
      const history = JSON.parse(localStorage.getItem('captureHistory') || '[]');
      history.unshift({
        title: entry.title,
        type: entry.type,
        url: entry.url || '',
        savedAt: new Date().toISOString()
      });
      localStorage.setItem('captureHistory', JSON.stringify(history.slice(0, MAX_HISTORY)));
      renderHistory();
    }

    function renderHistory() {
      const history = JSON.parse(localStorage.getItem('captureHistory') || '[]');
      const section = document.getElementById('history-section');
      const list = document.getElementById('history-list');

      if (history.length === 0) {
        section.classList.add('hidden');
        return;
      }

      section.classList.remove('hidden');
      list.innerHTML = history.map(h => {
        const ago = timeAgo(new Date(h.savedAt));
        return `<li>
          <span class="history-title">${escapeHtml(h.title)}</span>
          <span class="history-meta">${h.type} &middot; ${ago}</span>
        </li>`;
      }).join('');
    }

    function timeAgo(date) {
      const seconds = Math.floor((Date.now() - date.getTime()) / 1000);
      if (seconds < 60) return 'just now';
      if (seconds < 3600) return Math.floor(seconds / 60) + 'm ago';
      if (seconds < 86400) return Math.floor(seconds / 3600) + 'h ago';
      return Math.floor(seconds / 86400) + 'd ago';
    }

    function escapeHtml(str) {
      const div = document.createElement('div');
      div.textContent = str;
      return div.innerHTML;
    }
    ```

    Call `addToHistory()` in the form submit success handler (after 201 response).
    Call `renderHistory()` in the DOMContentLoaded handler.
