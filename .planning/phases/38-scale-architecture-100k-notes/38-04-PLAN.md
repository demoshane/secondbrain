       ```python
       CHUNK_SIZE = 1200      # characters per chunk
       CHUNK_OVERLAP = 200    # overlap between consecutive chunks
       CHUNK_THRESHOLD = 600  # notes shorter than this get single chunk

       def split_text_into_chunks(text: str, chunk_size: int = CHUNK_SIZE,
                                   overlap: int = CHUNK_OVERLAP) -> list[str]:
           """Split text into overlapping character-window chunks.
           Returns list of chunk strings. Short texts return [text]."""
           if len(text) <= chunk_size:
               return [text]
           chunks = []
           start = 0
           while start < len(text):
               end = min(start + chunk_size, len(text))
               chunks.append(text[start:end])
               if end == len(text):
                   break
               start += chunk_size - overlap
           return chunks

       def embed_chunks(text: str, provider: str | None = None,
                        batch_size: int = 32) -> list[tuple[str, bytes]]:
           """Split text into chunks and embed each. Returns [(chunk_text, embedding_blob), ...]."""
           chunks = split_text_into_chunks(text)
           if not chunks:
               return []
           blobs = embed_texts(chunks, provider=provider, batch_size=batch_size)
           return list(zip(chunks, blobs))
       ```

        chunk_count = 0
        for path, body, h in to_embed:
            if len(body) < CHUNK_THRESHOLD:

                blob = next(b for p, b, _ in zip(paths, blobs, hashes) if p == path)

                # Long note: split and embed each chunk
