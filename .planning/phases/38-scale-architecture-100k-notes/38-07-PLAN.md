       ```python
       SIMILARITY_THRESHOLD = 0.80  # cosine similarity above which notes are candidates

       def find_consolidation_candidates(conn, limit: int = 50) -> int:
           """Find similar notes with shared entities, write to consolidation_candidates.
           Returns count of new candidates found."""
           import json
           try:
               from engine.ann_index import knn_query, load_or_build_index
           except ImportError:
               logger.warning("hnswlib not available, skipping consolidation pass")
               return 0

           idx = load_or_build_index(conn=conn)
           if idx is None:
               return 0

           # Get all notes with embeddings
           rows = conn.execute(
               "SELECT path, people, tags FROM notes WHERE archived = 0"
           ).fetchall()

           # Get existing candidates to avoid re-processing
           existing = set()
           for r in conn.execute(
               "SELECT note_a, note_b FROM consolidation_candidates"
           ).fetchall():
               existing.add((r[0], r[1]))
               existing.add((r[1], r[0]))

           new_count = 0
           for path, people_json, tags_json in rows:
               if new_count >= limit:
                   break
               # Get embedding for this note
               emb_row = conn.execute(
                   "SELECT embedding FROM note_embeddings WHERE note_path=?", (path,)
               ).fetchone()
               if not emb_row or not emb_row[0]:
                   continue

               # KNN query
               neighbors = knn_query(emb_row[0], k=10, conn=conn)
               people_a = set(json.loads(people_json or "[]"))
               tags_a = set(json.loads(tags_json or "[]"))

               for neighbor_path, distance in neighbors:
                   if neighbor_path == path:
                       continue
                   similarity = 1.0 - distance
                   if similarity < SIMILARITY_THRESHOLD:
                       continue
                   pair = tuple(sorted([path, neighbor_path]))
                   if pair in existing:
                       continue

                   # Check shared entities
                   nb_row = conn.execute(
                       "SELECT people, tags FROM notes WHERE path=?", (neighbor_path,)
                   ).fetchone()
                   if not nb_row:
                       continue
                   people_b = set(json.loads(nb_row[0] or "[]"))
                   tags_b = set(json.loads(nb_row[1] or "[]"))
                   shared = list((people_a & people_b) | (tags_a & tags_b))
                   if not shared:
                       continue

                   # Insert candidate
                   try:
                       conn.execute(
                           """INSERT OR IGNORE INTO consolidation_candidates
                              (note_a, note_b, similarity, shared_entities)
                              VALUES (?, ?, ?, ?)""",
                           (pair[0], pair[1], similarity, json.dumps(shared)),
                       )
                       existing.add(pair)
                       new_count += 1
                   except Exception:
                       pass

           conn.commit()
           return new_count
       ```
