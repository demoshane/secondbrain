**engine/api.py — GET /tags endpoint** (per D-09):
```python
@app.get("/tags")
def list_tags():
    with get_connection() as conn:
        rows = conn.execute(
            "SELECT DISTINCT tag FROM note_tags ORDER BY tag"
        ).fetchall()
    tags = [r[0] for r in rows]
    if not tags:
        # Fallback: query JSON tags column if note_tags is empty (per Pitfall 6)
        rows = conn.execute("SELECT tags FROM notes WHERE tags IS NOT NULL AND tags != '[]'").fetchall()
        import json
        tag_set = set()
        for r in rows:
            try:
                tag_set.update(json.loads(r[0]))
            except (json.JSONDecodeError, TypeError):
                pass
        tags = sorted(tag_set)
    return jsonify({"tags": tags})
```

- Keyboard handling on the Input:
  - `ArrowDown`: increment highlightIndex (wrap at end)
  - `ArrowUp`: decrement highlightIndex (wrap at start)

const toggleDone = async (action: ActionItem) => {
  await fetch(`${getAPI()}/actions/${action.id}`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ done: !action.done }),

const assignTo = async (action: ActionItem, assigneePath: string) => {

}
