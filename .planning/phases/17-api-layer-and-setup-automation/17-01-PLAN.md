    Create engine/api.py:

    ```python
    """Local HTTP sidecar for Second Brain GUI.

    Exposes engine functions as HTTP endpoints on 127.0.0.1:37491.
    The GUI must call this API only — never import engine modules directly.
    """
    from flask import Flask, jsonify, request
    from flask_cors import CORS

    app = Flask(__name__)
    CORS(app, origins=["null", "file://*", "http://127.0.0.1:*"])
    ```

    Endpoints:

    GET /health: return jsonify({"status": "ok", "port": 37491})

    GET /notes:
      conn = get_connection()
      rows = conn.execute("SELECT path, title, type, created_at FROM notes ORDER BY created_at DESC").fetchall()
      conn.close()
      return jsonify({"notes": [dict(r) for r in rows]})

    POST /search:
      body = request.get_json(force=True) or {}
      query = body.get("query", "")
      conn = get_connection()
      results = search_notes(conn, query)
      conn.close()
      return jsonify({"results": results})

    GET /notes/&lt;path:note_path&gt;:
      Import Path at top.
      p = Path(note_path) if note_path.startswith("/") else Path("/") / note_path
      If not p.exists(): return jsonify({"error": "Not found"}), 404
      content = p.read_text(encoding="utf-8")
      return jsonify({"content": content, "path": str(p)})

    GET /actions:
      conn = get_connection()
      actions = list_actions(conn, done=False)
      conn.close()
      return jsonify({"actions": actions})

    def main():
      from waitress import serve
      serve(app, host="127.0.0.1", port=37491, threads=4)

    IMPORTANT: Do NOT call app.run() anywhere — Waitress only in main().
    IMPORTANT: search_notes returns list of dicts (already serializable); list_actions also returns list of dicts.
    If list_actions signature differs, check engine/intelligence.py at runtime; adapt accordingly.

    Run `uv add flask waitress flask-cors` before writing the file if deps not yet installed.
