    **Create `engine/passes/p4_actions.py` (per D-07 layer a, D-09):**

    ```python
    import re
    from engine.passes import ActionItem

    DEFAULT_MARKERS = ["TODO", "AP", "action:", "@owner", "Action Point"]

    def extract_keyword_actions(body: str, custom_markers: list[str] | None = None) -> list[ActionItem]:
        """Extract action items by keyword markers. No LLM, no DB."""
        from engine.config_loader import load_config
        from engine.paths import CONFIG_PATH

        if custom_markers is None:
            config = load_config(CONFIG_PATH)
            custom_markers = config.get("action_items", {}).get("custom_markers", [])

        all_markers = DEFAULT_MARKERS + custom_markers

        results = []
        for line in body.splitlines():
            stripped = line.strip()
            if not stripped:
                continue
            for marker in all_markers:
                # Check if line starts with marker (case-insensitive)
                if marker.startswith("@"):
                    # @owner pattern: extract owner name
                    m = re.match(r'@(\w+)\s+(.+)', stripped)
                    if m:
                        results.append(ActionItem(
                            text=m.group(2).strip(),
                            owner=m.group(1),
                            due_date=None,
                            source="keyword",
                        ))
                        break
                else:
                    pat = re.compile(re.escape(marker) + r'\s*[:\-]?\s*(.+)', re.IGNORECASE)
                    m = pat.match(stripped)
                    if m:
                        results.append(ActionItem(
                            text=m.group(1).strip(),
                            owner=None,
                            due_date=None,
                            source="keyword",
                        ))
                        break
        return results
    ```

    **Create `engine/passes/p5_assemble.py` (per D-08, D-12):**

    ```python
    def assemble(segments, conn, brain_root):
        """Run entity resolution on segments and populate person_stubs + existing_people.

        Does NOT write to DB — produces the blueprint for callers to persist.
        """
        from engine.segmenter import resolve_entities

        for result in segments:
            if not result.entities:
                continue
            resolution = resolve_entities(result.entities, conn, brain_root)
            result.person_stubs = resolution.get("new_stubs", [])
            result.existing_people = resolution.get("existing", [])

        return segments
    ```

    @app.put("/config/action-item-markers")
    def put_action_item_markers():
        import tomli_w
        from engine.config_loader import load_config
        from engine.paths import CONFIG_PATH
        data = request.get_json(force=True, silent=True) or {}
        markers = data.get("custom_markers", [])
        if not isinstance(markers, list):
            return jsonify({"error": "custom_markers must be a list"}), 400
        try:
            cfg = load_config(CONFIG_PATH)
