    Update `sb_capture_batch` in mcp_server.py:

    1. After the save loop completes (all notes saved), spawn a background daemon thread for intelligence hooks:
       ```python
       import threading

       _saved_paths = [s["path"] for s in succeeded]
       _brain_root = _BRAIN_ROOT

       def _run_batch_intelligence():
           for p in _saved_paths:
               try:
                   from engine.db import get_connection as _gc
                   _c = _gc()
                   try:
                       from engine.intelligence import check_connections, extract_action_items
                       check_connections(Path(p), _c, _brain_root)
                       _c.commit()
                   finally:
                       _c.close()
               except Exception as exc:
                   # Log to audit_log, never surface
                   try:
                       from engine.db import get_connection as _gc2
                       _c2 = _gc2()
                       _c2.execute(
                           "INSERT INTO audit_log (note_path, action, details, created_at) VALUES (?,?,?,?)",
                           (p, "intelligence_error", str(exc), datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ"))
                       )
                       _c2.commit()
                       _c2.close()
                   except Exception:
                       pass

               try:
                   from engine.db import get_connection as _gc3
                   _c3 = _gc3()
                   try:
                       # Read body from DB for action extraction
                       row = _c3.execute("SELECT body FROM notes WHERE path=?", (p,)).fetchone()
                       if row and row[0]:
                           extract_action_items(Path(p), row[0], "public", _c3)
                   finally:
                       _c3.close()
               except Exception as exc:
                   try:
                       from engine.db import get_connection as _gc4
                       _c4 = _gc4()
                       _c4.execute(
                           "INSERT INTO audit_log (note_path, action, details, created_at) VALUES (?,?,?,?)",
                           (p, "intelligence_error", str(exc), datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ"))
                       )
                       _c4.commit()
                       _c4.close()
                   except Exception:
                       pass

       threading.Thread(target=_run_batch_intelligence, daemon=True).start()
       ```
