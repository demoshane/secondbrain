```python
@app.route("/projects/<path:note_path>/meetings", methods=["POST"])
def link_meeting_to_project(note_path):
    """Link a meeting note to a project via relationships table."""
    data = request.get_json(force=True) or {}
    meeting_path = data.get("meeting_path", "").strip()
    if not meeting_path:
        abort(400, description="meeting_path is required")

    project_path = _resolve_note_path(note_path)
    conn = get_connection()
    try:
        # Verify project exists and is type=projects
        proj = conn.execute("SELECT path FROM notes WHERE path=? AND type='projects'", (project_path,)).fetchone()
        if not proj:
            abort(404, description="Project not found")
        # Verify meeting exists and is type=meeting
        mtg = conn.execute("SELECT path FROM notes WHERE path=? AND type='meeting'", (meeting_path,)).fetchone()
        if not mtg:
            abort(404, description="Meeting not found")
        # Check if relationship already exists
        existing = conn.execute(
            "SELECT id FROM relationships WHERE "
            "(source_path=? AND target_path=?) OR (source_path=? AND target_path=?)",
            (project_path, meeting_path, meeting_path, project_path),
        ).fetchone()
        if not existing:
            conn.execute(
                "INSERT INTO relationships (source_path, target_path, rel_type) VALUES (?, ?, ?)",
                (project_path, meeting_path, "linked"),
            )
            conn.commit()
        # Broadcast SSE
        _broadcast_sse("notes_changed", {"path": project_path})
    finally:
        conn.close()

    # Return updated project with linked_meetings
    # Re-fetch the project detail (reuse existing detail logic)
    return redirect(url_for("get_project_detail", note_path=note_path))
```
