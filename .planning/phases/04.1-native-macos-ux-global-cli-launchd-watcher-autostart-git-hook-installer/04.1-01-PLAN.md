find_uv() -> Path:
  shutil.which("uv") -> resolve() if found
  fallback: Path.home() / ".local" / "bin" / "uv"
  raise FileNotFoundError if neither exists

install_global_cli(repo_root: Path) -> None:
  subprocess.run(["uv", "tool", "install", "--editable", "--force", str(repo_root)], check=True)
  # Also call uv tool update-shell if ~/.local/bin not in os.environ.get("PATH","")

write_plist(sb_watch_bin: Path, repo_root: Path) -> Path:
  plist = {
    "Label": "com.secondbrain.watch",
    "ProgramArguments": [str(sb_watch_bin)],
    "WorkingDirectory": str(Path.home() / "SecondBrain"),
    "EnvironmentVariables": {"PATH": f"{Path.home()}/.local/bin:/usr/local/bin:/usr/bin:/bin", "HOME": str(Path.home())},
    "KeepAlive": True,
    "StandardOutPath": str(Path.home() / "Library" / "Logs" / "second-brain-watch.log"),
    "StandardErrorPath": str(Path.home() / "Library" / "Logs" / "second-brain-watch-error.log"),
  }
  plistlib.dump(plist, open(PLIST_PATH, "wb"))
  return PLIST_PATH

load_launchd_agent(plist_path: Path) -> None:
  uid = os.getuid()
  domain = f"gui/{uid}"
  subprocess.run(["launchctl", "bootout", f"{domain}/{PLIST_LABEL}"], capture_output=True)  # ignore error
  result = subprocess.run(["launchctl", "bootstrap", domain, str(plist_path)], capture_output=True, text=True)
  if result.returncode != 0: raise RuntimeError(result.stderr)

install_hook(repo_path: Path, hooks_dir: Path) -> None:
  result = subprocess.run(["/usr/bin/git", "-C", str(repo_path), "rev-parse", "--git-dir"], capture_output=True, text=True)
  if result.returncode != 0: raise ValueError(f"Not a git repo: {repo_path}")
  subprocess.run(["/usr/bin/git", "-C", str(repo_path), "config", "core.hooksPath", str(hooks_dir)], check=True)
