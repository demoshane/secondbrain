       def _recency_multiplier(created_at_str: str, half_life_days: int = 30) -> float:
           try:
               created = datetime.datetime.fromisoformat(str(created_at_str).rstrip("Z"))
               age_days = (datetime.datetime.utcnow() - created).days

           except Exception:
