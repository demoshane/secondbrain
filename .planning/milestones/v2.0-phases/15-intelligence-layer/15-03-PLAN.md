```python
def detect_git_context() -> str | None:
    try:
        result = subprocess.run(
            ["/usr/bin/git", "rev-parse", "--show-toplevel"],
            capture_output=True, text=True, timeout=5
        )
        if result.returncode == 0:
            return Path(result.stdout.strip()).name
    except (subprocess.TimeoutExpired, FileNotFoundError, OSError):
        pass
    return None
```

**find_similar** — KNN via sqlite-vec (wrap entirely in try/except):
```python
def find_similar(note_path: str, conn, threshold: float = 0.8, limit: int = 3) -> list[dict]:
    row = conn.execute(
        "SELECT embedding FROM note_embeddings WHERE note_path = ?", (note_path,)
    ).fetchone()
    if not row or not row[0]:
        return []
    query_blob = row[0]
    try:
        import sqlite_vec
        conn.enable_load_extension(True)
        sqlite_vec.load(conn)
    except Exception:
        return []
    try:
        rows = conn.execute(
            """
            SELECT ne.note_path, vec_distance_cosine(ne.embedding, ?) AS dist
            FROM note_embeddings ne
            WHERE ne.note_path != ?
              AND (1.0 - vec_distance_cosine(ne.embedding, ?)) >= ?
            ORDER BY dist
            LIMIT ?
            """,
            (query_blob, note_path, query_blob, threshold, limit),
        ).fetchall()
        return [{"note_path": r[0], "similarity": 1.0 - r[1]} for r in rows]
    except Exception:
        return []
```

```python
def check_connections(note_path: Path, conn, brain_root: Path) -> None:
    """Print connection suggestions for newly captured note. Best-effort."""
    if not budget_available(conn):
        return
    matches = find_similar(str(note_path.resolve()), conn)
    if not matches:
        return
    print("\nRelated notes found:")
    for m in matches:
        matched_path = Path(m["note_path"])
        print(f"  - {matched_path.stem}  (similarity: {m['similarity']:.2f})")
        # Append Related: [[stem]] to new note only — skip if PII (Phase 17 encryption safety)
        try:
            import frontmatter
            meta = frontmatter.load(str(note_path))
            if meta.get("content_sensitivity", "public") != "pii":
                _append_related_link(note_path, matched_path.stem)
        except Exception:
            pass
    consume_budget()
```

**_append_related_link** helper (from research):
```python
def _append_related_link(note_path: Path, matched_stem: str) -> None:
    """Append Related: [[stem]] to note — one-directional, best-effort."""
    try:
        existing = note_path.read_text(encoding="utf-8")
        link_line = f"\nRelated: [[{matched_stem}]]"
        if link_line.strip() not in existing:
            note_path.write_text(existing + link_line, encoding="utf-8")
    except OSError:
        pass
```

```python
def recap_main(argv=None) -> None:
    import argparse
    from engine.db import get_connection, init_schema
    parser = argparse.ArgumentParser(prog="sb-recap", description="Session recap for a context")
    parser.add_argument("name", nargs="?", default=None, help="Context name (optional; auto-detected from git)")
    args = parser.parse_args(argv)

    context_name = args.name or detect_git_context()
    if not context_name:
        print("No context detected — try sb-recap <name>")
        return

    conn = get_connection()
    init_schema(conn)
    rows = conn.execute(
        """
        SELECT title, body FROM notes
        WHERE tags LIKE ? OR people LIKE ? OR title LIKE ?
        ORDER BY updated_at DESC LIMIT 10
        """,
        (f"%{context_name}%", f"%{context_name}%", f"%{context_name}%"),
    ).fetchall()
    conn.close()

    if not rows:
        print(f"No notes found for context: {context_name}")
        return

    note_snippets = "\n".join(
        f"- {title}: {body[:200]}" for title, body in rows
    )
    try:
        from engine.paths import CONFIG_PATH
        adapter = _router.get_adapter("public", CONFIG_PATH)
        summary = adapter.generate(user_content=note_snippets, system_prompt=RECAP_SYSTEM_PROMPT)
        print(f"\nRecap for '{context_name}':\n{summary}")
    except Exception as e:
        print(f"[recap] Summarization failed: {type(e).__name__}")
```
