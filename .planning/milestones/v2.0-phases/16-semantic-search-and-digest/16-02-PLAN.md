    ```python
    def _rrf_merge(
        bm25_results: list[dict],
        vec_results: list[dict],
        k: int = 60,
        limit: int = 20,
    ) -> list[dict]:
        """Merge two ranked lists via Reciprocal Rank Fusion. Works on rank positions, NOT raw scores."""
        scores: dict[str, float] = {}
        all_items: dict[str, dict] = {}
        for rank, item in enumerate(bm25_results):
            p = item["path"]
            scores[p] = scores.get(p, 0.0) + 1.0 / (k + rank + 1)
            all_items[p] = item
        for rank, item in enumerate(vec_results):
            p = item["path"]
            scores[p] = scores.get(p, 0.0) + 1.0 / (k + rank + 1)
            all_items[p] = item
        ranked = sorted(scores.keys(), key=lambda p: scores[p], reverse=True)
        return [all_items[p] for p in ranked[:limit]]
    ```
